### 安装 vue-xlsx-table

```js
import Vue from "vue"
import App from "./App.vue"

import vueXlsxTable from "vue-xlsx-table"

Vue.use(vueXlsxTable)

new Vue({
  el: "#app",
  render: h => h(App)
})
```

```html
<template>
  <div id="app">
    <h1>Vue-xlsx-table</h1>
    <p>Not need upload, view xlsx or xls file in your browser, Supported by js-xlsx.</p>
    <vue-xlsx-table @on-select-file="handleSelectedFile">Select one file</vue-xlsx-table>
  </div>
</template>

<script>
  export default {
    name: "app",
    methods: {
      handleSelectedFile(convertedData) {
        console.log(convertedData)
        window.alert(JSON.stringify(convertedData))
      }
    }
  }
</script>
```
